import paho.mqtt.client as mqtt

# Token Ubidots
UBIDOTS_TOKEN = "BBUS-vFMys8yflUN2qQUBLphNia3F5gmB89"

# Configuration MQTT
MQTT_BROKER = "industrial.api.ubidots.com"
MQTT_PORT = 1883
MQTT_USERNAME = UBIDOTS_TOKEN
MQTT_PASSWORD = ""

# Topic à écouter
TOPIC = "/v1.6/devices/raspberry/temperature/lv"

# Fonction appelée quand la connexion est établie
def on_connect(client, userdata, flags, rc, properties=None):
    if rc == 0:
        print("✅ Connecté à Ubidots avec succès!")
        client.subscribe(TOPIC)
        print(f"📡 Abonné au topic: {TOPIC}")
    else:
        print(f"❌ Échec de connexion, code: {rc}")

# Fonction appelée quand un message arrive
def on_message(client, userdata, msg):
    print(f"📨 Message reçu sur {msg.topic} : {msg.payload.decode()}")

# Création du client
client = mqtt.Client(mqtt.CallbackAPIVersion.VERSION2)

# Assigner les fonctions
client.on_connect = on_connect
client.on_message = on_message

# Authentification
client.username_pw_set(MQTT_USERNAME, MQTT_PASSWORD)

# Connexion au broker
print("🔄 Connexion à Ubidots...")
client.connect(MQTT_BROKER, MQTT_PORT, 60)

print("⏳ En attente de messages...")

# Boucle infinie
client.loop_forever()
